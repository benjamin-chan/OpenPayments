source("getData.r")
source("library.r")

Map at the state level.

DAgg <- D[,
          list(value = sum(Total_Amount_of_Payment_USDollars) / 1E6,
               region = Recipient_State),
          list(Recipient_State)]
require(choroplethr)
choroplethr(DAgg, 
            lod="state", 
            title="General + Research Payments")
